   434‚Üí# --- Build third derivative tensor W_IJK ---
   435‚Üí# The superpotential has third derivatives from:
   436‚Üí# (1) X * det M: d^3(X det M)/dX dM^a_b dM^c_d = d^2(det M)/dM^a_b dM^c_d
   437‚Üí#                d^3(X det M)/dM^a_i dM^b_j dM^c_k = X * eps_{abc} eps_{ijk}
   438‚Üí# (2) lambda X Hu0 Hd0: d^3/dX dHu0 dHd0 = lambda
   439‚Üí# (3) -lambda X Hu+ Hd-: d^3/dX dHu+ dHd- = -lambda
   440‚Üí# (4) No third derivatives from Yukawa (bilinear) or mass terms (linear)
   441‚Üí
   442‚Üídef levi_civita_sign(p):
   443‚Üí    p = list(p)
   444‚Üí    sign = 1
   445‚Üí    for i in range(len(p)):
   446‚Üí        while p[i] != i:
   447‚Üí            sign *= -1
   448‚Üí            j = p[i]
   449‚Üí            p[i], p[j] = p[j], p[i]
   450‚Üí    return sign
